def convert_price_format(price_str):
    """ Convertit une offre de prix en valeur entiÃ¨re """
    print(f"[DEBUG] Raw price input: {price_str}")  # âœ… Ajout Debug
    if isinstance(price_str, int):  # âœ… VÃ©rifie si c'est dÃ©jÃ  un entier
        return price_str

    price_str = price_str.lower().replace("qar", "").replace(",", "").strip()

    # âœ… VÃ©rifie si c'est un format "5K6" ou "6.5k"
    match = re.search(r'(\d+)\s?[kK](\d+)?', price_str)
    if match:
        num_part = int(match.group(1)) * 1000
        decimal_part = int(match.group(2)) * 100 if match.group(2) else 0
        result = num_part + decimal_part
        print(f"[DEBUG] Parsed price: {result} QAR")  # âœ… Ajout Debug
        return result

    # âœ… VÃ©rifie un format "6.5k" ou "5.5k"
    match = re.search(r'(\d+)\.?(\d+)?\s?[kK]', price_str)
    if match:
        num_part = int(match.group(1)) * 1000
        decimal_part = int(match.group(2)) * 100 if match.group(2) else 0
        result = num_part + decimal_part
        print(f"[DEBUG] Parsed price: {result} QAR")  # âœ… Ajout Debug
        return result

    # âœ… VÃ©rifie un format numÃ©rique classique
    match = re.search(r'\b\d+\b', price_str)
    if match:
        result = int(match.group(0))
        print(f"[DEBUG] Parsed price: {result} QAR")  # âœ… Ajout Debug
        return result

    print(f"[DEBUG] Failed to parse price: {price_str}")  # âœ… Debug si la conversion Ã©choue
    return None

def correct_spelling(user_input):
    """ Corrige les fautes de frappe en comparant avec les mots-clÃ©s disponibles """
    all_keywords = [word for sublist in keywords.values() for word in sublist]
    corrected_words = []
    
    for word in user_input.split():
        close_match = difflib.get_close_matches(word, all_keywords, n=1, cutoff=0.8)
        corrected_words.append(close_match[0] if close_match else word)
    
    return " ".join(corrected_words)

# ðŸ“Œ Fonction de correspondance approximative
def fuzzy_match(user_input, category_keywords):
    """ VÃ©rifie si l'entrÃ©e utilisateur correspond approximativement Ã  un des mots-clÃ©s """
    best_score = 0
    best_match = None
    adaptive_threshold = 80  # Seuil par dÃ©faut

    for keyword in category_keywords:
        score = fuzz.partial_ratio(user_input, keyword)

        # Si le mot-clÃ© est long, baisser le seuil pour mieux reconnaÃ®tre les erreurs
        if len(keyword) > 10:
            adaptive_threshold = 75  

        if score > best_score:
            best_score = score
            best_match = keyword

    return best_score >= adaptive_threshold

def convert_k_notation(price_str):
    """ Convertit des valeurs comme 10K en 10000 et 5.6K en 5600 """
    price_str = price_str.upper().replace("QAR", "").replace(",", "").strip()
    
    pattern = r'(\d+(\.\d+)?)K'  # Gestion des valeurs avec un point dÃ©cimal (ex: 5.6K)
    matches = re.findall(pattern, price_str)

    for match in matches:
        num_part = float(match[0]) * 1000
        converted_value = str(int(num_part))
        price_str = price_str.replace(match[0] + "K", converted_value)

    return price_str
